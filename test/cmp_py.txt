import numpy as np
Width,Height = 1000,1000

def gen() :
     X = np.array([1/4,0,0])
     Y = np.array([0,1/4,0])
     Z = np.array([0,0,1/4])
 
     W = np.array(np.linspace(0, Width-1, Width)).reshape((Width, 1))
     H = np.array(np.linspace(Height-1, 0, Height)).reshape((Height, 1, 1))
     return X*W + (Y*H+Z)


timeit.Timer(gen).timeit(10)
